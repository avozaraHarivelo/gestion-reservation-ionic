import Konva from 'konva';

export class MonthCell {
    private cellWidth: number;
    private cellHeight: number;
    private monthText!: Konva.Text;
    private cell!: Konva.Rect;

    constructor(private stage: Konva.Stage, private monthLayer: Konva.Layer, private positionX: number, private positionY: number, private monthName: string) {
        this.cellWidth = 100; // Set the appropriate width
        this.cellHeight = 50; // Set the appropriate height
        
        this.createMonthCell();
    }

    private createMonthCell() {
        this.cell = new Konva.Rect({
            width: this.cellWidth,
            height: this.cellHeight,
            fill: 'lightblue', // Set the appropriate fill color
            stroke: 'grey',
            strokeWidth: 0.5,
            className: 'month-cell'
        });

        this.cell.position({
            x: this.positionX,
            y: this.positionY
        });

        this.monthText = new Konva.Text({
            text: this.monthName,
            width: this.cellWidth,
            height: this.cellHeight,
            align: 'center',
            verticalAlign: 'middle',
            fontSize: 14,
            fill: 'black'
        });

        this.monthText.position({
            x: this.positionX,
            y: this.positionY
        });

        this.monthLayer.add(this.cell); // Add to the monthLayer
        this.monthLayer.add(this.monthText); // Add to the monthLayer
    }

    // Other methods for handling interactions, animations, etc.
}
